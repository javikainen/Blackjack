420      */
421     public void startGame() {
422         nextPhase(Phase.BET);
423     }
424 
425     private void nextPhase(Phase phase) {
426         this.phase = phase;
427         setButtons();
428         switch (this.phase) {
429             case BET:
430                 break;
431             case DEAL:
432                 dealPhase();
433                 break;
434             case PLAY:
435                 playPhase();
436                 break;
437             case USER_TURN:
438                 break;
439             case PAY:
440                 payPhase();
441         }
442     }

444     private void setButtons() {
445         betAdd5Button.setEnabled(false);
446         betAdd10Button.setEnabled(false);
447         dealButton.setEnabled(false);
448         hitButton.setEnabled(false);
449         standButton.setEnabled(false);
450         zeroBetButton.setEnabled(false);
451         switch (this.phase) {
452             case BET:
453                 betAdd5Button.setEnabled(true);
454                 betAdd10Button.setEnabled(true);
455                 dealButton.setEnabled(true);
456                 zeroBetButton.setEnabled(true);
457                 break;
458             case USER_TURN:
459                 hitButton.setEnabled(true);
460                 standButton.setEnabled(true);
461             default:
462         }
463     }

465     private void displayHand(JPanel area, Hand hand) {
466         area.removeAll();
467         if (area == dealerCardPanel && phase == Phase.DEAL) {
468             area.add(new ImagePanel(cardImages.getCardBack(cardColor)));
469             area.add(new ImagePanel(cardImages.getCardImage(hand.getContents().get(1))), 0);
470         } else {
471             for (Card card : hand.getContents()) {
472                 area.add(new ImagePanel(cardImages.getCardImage(card)), 0);
473             }
474         }
475         area.validate();
476     }

478     private JPanel getPlayArea(Player player) {
479         switch (player.getID()) {
480             case 0:
481                 return AICardPanel2;
482             case 1:
483                 return playerCardPanel;
484             case 2:
485                 return AICardPanel1;

506     private void dealPhase() {
507         for (Player player : gameLogic.getPlayers()) {
508             getHandValueLabel(player).setText("");
509             JPanel area = getPlayArea(player);
510             area.setEnabled(false);
511         }
512         gameLogic.deal();
513         for (Player player : gameLogic.getPlayers()) {
514             JPanel area = getPlayArea(player);
515             area.setEnabled(true);
516             displayHand(area, player.getHand());
517         }
518         nextPhase(Phase.PLAY);
519     }
520 
521     private void playPhase() {
522         while (true) {
523             player = gameLogic.getNextPlayer();
524             if (player == null) {
525                 nextPhase(Phase.PAY);
526                 break;
527             } else if (player.isAI()) {
528                 gameLogic.playAIHand(player);
529             } else {
530                 nextPhase(Phase.USER_TURN);
531                 break;
532             }
533             displayHand(getPlayArea(player), player.getHand());
534         }
535     }
536 
537     private void payPhase() {
538         for (Player player : gameLogic.getPlayers()) {
539             String result;
540             if (player.getHand().isBlackJack()) {
541                 result = "Blackjack!";
542             } else if (player.getHand().isBust()) {
543                 result = "Bust!";
544             } else {
545                 result = Integer.toString(player.getHand().getValue());
546             }
547             getHandValueLabel(player).setText(result);
548         }
549         gameLogic.payWinnings();
550         playerMoneylabel.setText("Player money: " + humanPlayer.getMoney());
551         nextPhase(Phase.BET);
552     }
