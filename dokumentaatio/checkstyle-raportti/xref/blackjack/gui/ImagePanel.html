1   package blackjack.gui;
2   
3   import java.awt.Dimension;
4   import java.awt.Graphics;
5   import java.awt.Image;
6   import javax.swing.JPanel;
7   
8   /**
9    * ImagePanel is used as a container for an image of a single card.
10   *
11   * @author Jari Avikainen
12   */
13  class ImagePanel extends JPanel {
14  
15      private final Image img;
16  
17      /**
18       * Generates a new ImagePanel using the image given as a parameter.
19       * 
20       * @param img The image to be shown in the ImagePanel.
21       */
22      public ImagePanel(Image img) {
23          this.img = img;
24          Dimension size = new Dimension(img.getWidth(null), img.getHeight(null));
25          setPreferredSize(size);
26          setMinimumSize(size);
27          setMaximumSize(size);
28          setSize(size);
29          setLayout(null);
30      }
31  
32      @Override
33      public void paintComponent(Graphics g) {
34          g.drawImage(img, 0, 0, null);
35      }
36  }
