9   public class Dealer {
10  
11      Hand hand;
12      Shoe shoe;
13      ArrayList<Player> players;
14  
15      public Hand getHand() {
16          return this.hand;
17      }
18  
19      public void getBets() {
20          for (Player player : players) {
21              player.setBet();
22          }
23      }
24  
25      public void deal() {
26          for (int i = 0; i < 2; i++) {
27              for (Player player : players) {
28                  if (!shoe.hasCards()) {
29                      shoe.shuffle();
30                  }
31                  player.addCard(shoe.getCard());
32              }
33              this.hand.add(shoe.getCard());
34          }
35      }
36  
37      public void resolveRound() {
38          for (Player player : players) {
39              resolveHand(player);
40          }
41          this.hand = new Hand();
42      }
43  
44      public void resolveHand(Player player) {
45          Choice playerChoice = player.getChoice();
46          boolean playerWin = true;
47          if (playerWin) {
48              player.addMoney(player.getCurrentBet());
49          } else {
50              player.clearBet();
51          }
52          player.clearHand();
53      }
54  
55      public void play() {
56          while (true) {
57              getBets();
58              deal();
59              resolveRound();
60          }
61      }
62  
63  }
