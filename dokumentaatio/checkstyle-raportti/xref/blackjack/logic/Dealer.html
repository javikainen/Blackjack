13      Hand hand;

20          this.shoe = new Shoe(1);
21      }
22  
23      public Hand getHand() {
24          return this.hand;
25      }
26  
27      public void deal() {
28          for (int i = 0; i < 2; i++) {
29              for (Player player : players) {
30                  player.addCard(shoe.getCard());
31              }
32              this.hand.add(shoe.getCard());
33          }
34      }
