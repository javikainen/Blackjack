3   import java.util.ArrayList;
4   
5   /**
6    * This class is the main game logic class. BlackjackLogic keeps track of all
7    * players, the shoe etc. and takes care of dealing cards, deciding the winners
8    * and so on.
9    *
10   * @author Jari Avikainen
11   */
12  public class BlackjackLogic {
13  
14      private ArrayList<Player> players;
15      private int nextPlayer;
16      private final Shoe shoe;
17      private Dealer dealer;
18  
19      /**
20       * Generate a new instance of BlackjackLogic using given parameters as the
21       * settings for the game. Players are generated and added to ArrayList
22       * players Players according to the order in which they will play their
23       * hands.
24       *
25       * @param numAIs The amount of AI players participating in the game. Allowed
26       * values are 0, 1 and 2.
27       * @param numDecks Number of 52-card decks to be shuffled together into the
28       * shoe.
29       * @param startingMoney Amount of starting money given to the players.
30       */
31      public BlackjackLogic(int numAIs, int numDecks, int startingMoney) {
32          if (numAIs < 0 || numAIs > 2 || numDecks < 1 || startingMoney <= 0) {
33              throw new IllegalArgumentException();
34          }
35          players = new ArrayList<>();
36          dealer = new Dealer("Dealer", 3, 0);
37          if (numAIs == 2) {
38              players.add(new AIPlayer("AI 1", 0, startingMoney));
39          }
40          players.add(new HumanPlayer("Human", 1, startingMoney));
41          if (numAIs > 0) {
42              players.add(new AIPlayer("AI " + numAIs, 2, startingMoney));
43          }
44          players.add(dealer);
45          shoe = new Shoe(numDecks);
46          nextPlayer = 0;
47      }
48  
49      /**
50       * Returns the hand belonging to the given player.
51       *
52       * @param player The player whose hand will be returned.
53       * @return The current hand of cards belonging to the given player.
54       */
55      public Hand getPlayerHand(Player player) {
56          return player.getHand();
57      }
58  
59      /**
60       * Requests each of the players to set their bet for the next round. The bet
61       * chosen by the user is passed to the players as an argument.
62       *
63       * @param userBet The new bet chosen by the user.
64       */
65      public void adjustBets(int userBet) {
66          for (Player player : players) {
67              player.adjustBet(userBet);
68          }
69      }
70  
71      /**
72       * Deals two cards to each of the players and the dealer.
73       *
74       */
75      public void deal() {
76          for (int i = 0; i < 2; i++) {
77              for (Player player : players) {
78                  player.addCard(shoe.getCard());
79              }
80          }
81          nextPlayer = 0;
82      }
83  
84      /**
85       * Deals one card to the player given as a parameter.
86       *
87       * @param player The Player to which the new card is dealt.
88       */
89      public void dealCard(Player player) {
90          player.addCard(shoe.getCard());
91      }
92  
93      /**
94       * Returns the Player whose turn it is to play.
95       *
96       * @return Player whose turn it is at the moment or null, if all the players
97       * (including the dealer) have played their turn.
98       */
99      public Player getNextPlayer() {
100         Player player;
101         if (nextPlayer >= players.size()) {
102             player = null;
103             nextPlayer = 0;
104         } else {
105             player = players.get(nextPlayer);
106             nextPlayer++;
107         }
108         return player;
109     }
110 
111     /**
112      * Instructs the AI player given as a parameter to play its turn. Throws an
113      * IllegalArgumentException if the Player given is not an AI.
114      *
115      * @param player The Player that should play its turn.
116      */
117     public void playAIHand(Player player) {
118         if (player.isAI()) {
119             player.playHand(shoe);
120         } else {
121             throw new IllegalArgumentException("playAIHand() argument is not an AI player!");
122         }
123     }
124 
125     public ArrayList<Player> getPlayers() {
126         return players;
127     }
128 
129     /**
130      * Returns the dealer's hand.
131      *
132      * @return The dealer's hand.
133      */
134     public Hand getDealerHand() {
135         return dealer.getHand();
136     }
137 
138     /**
139      * Goes through all of the players, evaluates their hands compared to the
140      * dealer, and adjusts the player's money according to the results.
141      *
142      * Hands are cleared after resolving them.
143      *
144      */
145     public void payWinnings() {
146         Hand dealerHand = dealer.getHand();
147         for (Player player : players) {
148             Hand playerHand = player.getHand();
149             if (player.isDealer()) {
150                 player.clearHand();
151                 break;
152             }
153             if (playerHand.isBust()) {
154                 player.takeMoney(player.getCurrentBet());
155             } else if (dealerHand.isBlackJack() && !playerHand.isBlackJack()) {
156                 player.takeMoney(player.getCurrentBet());
157             } else if (playerHand.isBlackJack() && !dealerHand.isBlackJack()) {
158                 player.addMoney(player.getCurrentBet() * 3 / 2);
159             } else if (dealerHand.isBust()) {
160                 player.addMoney(player.getCurrentBet());
161             } else if (playerHand.getValue() > dealerHand.getValue()) {
162                 player.addMoney(player.getCurrentBet());
163             } else if (playerHand.getValue() < dealerHand.getValue()) {
164                 player.takeMoney(player.getCurrentBet());
165             }
166             player.clearHand();
167         }
168     }
