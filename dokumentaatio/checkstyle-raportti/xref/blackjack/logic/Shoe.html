1   package blackjack.logic;

3   import java.util.Collections;
4   import java.util.LinkedList;
5   import java.util.Random;
6   
7   /**
8    * This class contains the shoe (the deck of cards) used by the dealer.
9    *
10   * @author Jari Avikainen
11   */
12  public class Shoe {
13  
14      private LinkedList<Card> contents;
15  
16      private final int decks;
17  
18      /**
19       * Fills shoe with the provided number of decks. The decks are added to the
20       * shoe, which is then shuffled.
21       *
22       * @param decks The number of 52-card decks to be used in the game.
23       */
24      public Shoe(int decks) {
25          this.decks = decks;
26          this.refill();
27      }
28  
29      /**
30       * Refills the shoe with the amount of decks provided in the constructor and
31       * shuffles it afterwards. The cards possibly remaining in the shoe are
32       * discarded.
33       */
34      public void refill() {
35          this.contents = new LinkedList<>();
36          for (int i = 0; i < this.decks; i++) {
37              for (Suit suit : Suit.values()) {
38                  for (int j = 1; j <= 13; j++) {
39                      this.contents.add(new Card(suit, j));
40                  }
41              }
42          }
43          Collections.shuffle(this.contents, new Random());
44      }
45  
46      /**
47       * The method returns true if the shoe is empty.
48       *
49       * @return true if the shoe has no cards left.
50       */
51      public boolean hasCards() {
52          return !this.contents.isEmpty();
53      }
54  
55      /**
56       * The method removes the next card from the shoe and returns it. If shoe
57       * has no cards left, it is refilled first.
58       *
59       * @return The card taken from the shoe
60       */
61      public Card getCard() {
62          if (!hasCards()) {
63              refill();
64          }
65          return this.contents.pop();
66      }
67  
68  }
