3   /**
4    * This class is the AI version of the Player.
5    *
6    * @author Jari Avikainen
7    */
8   public class AIPlayer extends Player {
9   
10      /**
11       * Generates a new AIPlayer using the given parameters.
12       *
13       * @param name The name of the player
14       * @param id The id used for connecting the player to a proper playing are
15       * in the UI
16       * @param money The money player has when starting the game
17       */
18      public AIPlayer(String name, int id, int money) {
19          super(name, id, money);
20      }
21  
22      @Override
23      public boolean isAI() {
24          return true;
25      }
26  
27      @Override
28      public void playHand(Shoe shoe) {
29          while (!handIsGoodEnough()) {
30              getHand().add(shoe.getCard());
31          }
32      }
33  
34      private boolean handIsGoodEnough() {
35          return (getHand().getValue() > 17 || getHand().getValue() == 17 && !getHand().containsAce());
36      }
37  
38      @Override
39      public void adjustBet(int newBet) {
40          super.adjustBet(10);
41      }
42  
43  }
